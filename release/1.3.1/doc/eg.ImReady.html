import ImReady from "@egjs/imready";

const im = new ImReady(); // umd: eg.ImReady
im.check(document.querySelectorAll("img")).on({
  preReadyElement: e => {
    // 1, 3
    // 2, 3
    // 3, 3
    console.log(e.preReadyCount, e.totalCount),
  },
});
